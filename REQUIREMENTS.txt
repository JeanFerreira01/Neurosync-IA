━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Clonar o repositório
     git clone https://github.com/JeanFerreira01/Neurosync-IA.git
     cd Neurosync-IA

  2. Criar e ativar o venv (PowerShell)
     python -m venv venv
     .\venv\Scripts\Activate.ps1

  3. Instalar dependências Python
     pip install -r backend/requirements/development.txt

  4. Configurar variáveis de ambiente
     copy backend\.env.example backend\.env
     (editar backend\.env com as chaves reais)

  5. Aplicar migrations do banco
     cd backend
     python manage.py migrate

  6. Criar superusuário (opcional)
     python manage.py createsuperuser

  7. Instalar dependências Node
     cd ..\frontend
     npm install

  8. Subir os servidores (dois terminais)

     Terminal 1 — Backend:
       cd backend
       $env:DJANGO_SETTINGS_MODULE="config.settings.development"
       ..\venv\Scripts\python.exe manage.py runserver
       → http://localhost:8000

     Terminal 2 — Frontend:
       cd frontend
       npm run dev
       → http://localhost:5173
